#include <iostream>

using namespace std;

bool isPrime(int);

int main()
{
	int number;
	
	do {
		cout << "\nEnter a number and I will tell you if it's prime. (0 to end): "; // zero sentinel value
		cin >> number;
		
		
		while (number < 0) {
			cout << "Please enter a number greater than zero.";
			cin >> number;
		}
		
		// have a good value to test
		if (isPrime(number)) {
			cout << "\nThis is a prime number.\n";
		} else {
			cout << "\nThis number is not a prime number.\n";
		}
	} while (number > 0);
	
	return 0;
}

// test for prime number
bool isPrime(int num) {
	bool prime = true;
	int testNum;
	
	switch (num) {
	case 1:
	case 2:
	case 3:
		break;
	default:
		// eliminate even numbers
		if (number % 2 == 0) {
			prime = false; // this is an even number
		} else {
			testNum = 3; // try dividing by three
			while (prime && testNum < (num/2+1)) {
				// test thisNumber to see if it's a divisor
				if(num % testNum == 0) {
					prime = false; // this number has a divisor
				} else {
					testNum += 2;
				}
			}
		
		}
	
	}
	return prime;
}
